        const words = str.split(' ');
        let lines = [];
        let current_line = words[0];

        for (let i = 1; i < words.length; i++) {
            if (current_line.length + words[i].length + 1 < max_width) {
                current_line += ' ' + words[i];
            } else {
                lines.push(current_line);
                current_line = words[i];
            }
        }
        lines.push(current_line);
        return lines;
    }

    };
